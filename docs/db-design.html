3. CREATE TABLE (SQLite)

-- players
CREATE TABLE players (
  id         TEXT    PRIMARY KEY,
  name       TEXT    NOT NULL,
  money      INTEGER NOT NULL DEFAULT 0,
  created_at TEXT    NOT NULL,
  updated_at TEXT    NOT NULL
);

-- pets
CREATE TABLE pets (
  id         TEXT    PRIMARY KEY,
  player_id  TEXT    NOT NULL REFERENCES players(id),
  name       TEXT    NOT NULL,
  species    TEXT    NOT NULL,
  level      INTEGER NOT NULL DEFAULT 1,
  happiness  INTEGER NOT NULL DEFAULT 80,
  hunger     INTEGER NOT NULL DEFAULT 60,
  created_at TEXT    NOT NULL
);

-- inventory
CREATE TABLE inventory (
  id          TEXT    PRIMARY KEY,
  player_id   TEXT    NOT NULL REFERENCES players(id),
  item_id     TEXT    NOT NULL,
  name        TEXT    NOT NULL,
  sell_price  INTEGER NOT NULL,
  obtained_at TEXT    NOT NULL
);

-- sell_history
CREATE TABLE sell_history (
  id           TEXT    PRIMARY KEY,
  player_id    TEXT    NOT NULL REFERENCES players(id),
  total_amount INTEGER NOT NULL,
  item_count   INTEGER NOT NULL,
  sold_at      TEXT    NOT NULL
);

-- インデックス
CREATE INDEX idx_pets_player       ON pets(player_id);
CREATE INDEX idx_inventory_player  ON inventory(player_id);
CREATE INDEX idx_sell_history_player ON sell_history(player_id);
-- ランキング用
CREATE INDEX idx_players_money     ON players(money DESC);
